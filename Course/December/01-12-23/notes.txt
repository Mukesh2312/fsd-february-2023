useState:
//To maintain any data dyanmically we need to use state
example:
import {useState} from 'react';

function App(){
    const[getName,setName]=useState('akash');

    const getDisplay=()=>{
        setName("suresh");
    }

    return (<div>
      {getName}
    </div>)
}

How to use Events in reactjs 
function App(){

    const getDisplay=()=>{
        setName("suresh");
    }

    return (<div>
